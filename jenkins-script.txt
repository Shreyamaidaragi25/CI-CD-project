pipeline {
    agent any

    stages {

        stage('Git Checkout') {
            steps {
                git branch: 'main',
                url: 'YOUR_GITHUB_REPO'
            }
        }

        stage('Compile') {
            steps {
                sh 'mvn compile'
            }
        }

        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }

        stage('Package') {
            steps {
                sh 'mvn package'
            }
        }

        stage('Sonar Analysis') {
            steps {
                withSonarQubeEnv('sonar-server') {
                    sh 'mvn sonar:sonar'
                }
            }
        }
stage('exposing to internet') {
    steps {
        sh '''
        cp target/*.war /home/ubuntu/tomcat/webapps/jpetstore.war

        cd /home/ubuntu/tomcat/bin

        ./shutdown.sh || true
        sleep 10

        ./startup.sh
        '''
    }
}

    }
}
